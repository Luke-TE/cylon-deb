#!/bin/bash
#Module containing functions called from cylon package
#Functions: SystemCleanFunc 

#FUNCTION HEADER
# NAME :  SystemCleanFunc 
# DESCRIPTION: Function deletes trash can and download folder.

function SystemCleanFunc
{
clear
local choiceFsVar=""
msgFunc blue "Delete files/folders. options:-"
local optionsFsArr=("Delete Trash folder" \
"Delete Download folder" "Delete Cylon output folder $DEST3" "Return")
	select choiceFsVar in "${optionsFsArr[@]}"
	do
	case "$choiceFsVar" in  
		
		"${optionsFsArr[0]}")  msgFunc green "Deleting Trash folder "
			 rm -rvf "$HOME"/.local/share/Trash/*
		;;
		
		"${optionsFsArr[1]}")  msgFunc green "Deleting Download folder "
			 rm -rvf "$HOME"/Downloads/*
		;;
		
		"${optionsFsArr[2]}")  msgFunc green "Deleting Cylon output folder $DEST3"
			msgFunc norm "Do you want proceed ?"
			if msgFunc yesno 
			then
				rm -rvf "$DEST3"
				#remake the path for the program output DEST3
				mkdir -p "$DEST3"
			fi
		;;
		
		*)  #exit  
			msgFunc green "Done!"	
		return
		;;
	esac
	break
	done

}
